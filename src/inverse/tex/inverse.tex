\newcommand{\init}{(\Gamma_0^-,C_0^+)}

\infer[Atom]{P \vdash P}{P \leq^+ \init & P \leq^- \init}

\infer[Contract]{\Gamma,A \vdash B}{\Gamma,A,A\vdash B}

\infer[ArrowR1]{\Gamma\vdash A \to B}{\Gamma,A \vdash B & (A\to B \leq^+ \init)}
